---
openapi: 'GET /v1beta/domains'
---

This endpoint retrieves a list of all domains associated with your account.

### Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful.
</ResponseField>
<ResponseField name="data" type="array">
  An array of domain objects.
  <ResponseField name="id" type="integer">
    The unique identifier for the domain.
  </ResponseField>
  <ResponseField name="url" type="string">
    The URL of the domain.
  </ResponseField>
  <ResponseField name="uuid" type="string">
    The universally unique identifier for the domain.
  </ResponseField>
  <ResponseField name="monitor" type="object">
    An object containing the monitor status.
    <ResponseField name="uptime_status" type="string">
      The current uptime status (e.g., `up`, `down`).
    </ResponseField>
    <ResponseField name="certificate_status" type="string">
      The status of the SSL certificate (e.g., `valid`, `expiring_soon`).
    </ResponseField>
    <ResponseField name="certificate_expiration_date" type="string">
      The expiration date of the SSL certificate.
    </ResponseField>
  </ResponseField>
  <ResponseField name="pivot" type="object">
    An object containing user-specific information about the domain.
    <ResponseField name="user_domain_name" type="string">
      The custom name given to the domain by the user.
    </ResponseField>
  </ResponseField>
</ResponseField>
